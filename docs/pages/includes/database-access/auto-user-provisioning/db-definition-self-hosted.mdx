Next, enable the database admin on the Teleport database configuration:

<Tabs>
<TabItem label="Static config">
```yaml
db_service:
  enabled: "yes"
  databases:
  - name: "example"
    protocol: "{{ protocol }}"
    uri: "{{ uri }}"
    admin_user:
      name: "teleport-admin"
```
</TabItem>
<TabItem label="Dynamic resource">
```yaml
kind: db
version: v3
metadata:
  name: example
spec:
  protocol: "{{ protocol }}"
  uri: "{{ uri }}"
  admin_user:
    name: "teleport-admin"
```
</TabItem>
</Tabs>
